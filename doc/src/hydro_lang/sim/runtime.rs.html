1use std::cell::RefCell;
2use std::collections::VecDeque;
3use std::rc::Rc;
4
5use bolero::ValueGenerator;
6use bolero::generator::bolero_generator::driver::object::Borrowed;
7use tokio::sync::mpsc::UnboundedSender;
8
9pub trait SimHook {
10    fn current_decision(&self) -> Option<bool>;
11    fn can_make_nontrivial_decision(&self) -> bool;
12    fn autonomous_decision<'a>(
13        &mut self,
14        driver: &mut Borrowed<'a>,
15        force_nontrivial: bool,
16    ) -> bool;
17    fn release_decision(&mut self);
18}
19
20pub struct StreamHook<T> {
21    pub input: Rc<RefCell<VecDeque<T>>>,
22    pub to_release: Option<Vec<T>>,
23    pub output: UnboundedSender<T>,
24}
25
26impl<T> SimHook for StreamHook<T> {
27    fn current_decision(&self) -> Option<bool> {
28        self.to_release.as_ref().map(|v| !v.is_empty())
29    }
30
31    fn can_make_nontrivial_decision(&self) -> bool {
32        !self.input.borrow().is_empty()
33    }
34
35    fn autonomous_decision<'a>(
36        &mut self,
37        driver: &mut Borrowed<'a>,
38        force_nontrivial: bool,
39    ) -> bool {
40        let mut current_input = self.input.borrow_mut();
41        let count = ((if force_nontrivial { 1 } else { 0 })..=current_input.len())
42            .generate(driver)
43            .unwrap();
44        self.to_release = Some(current_input.drain(0..count).collect());
45        count > 0
46    }
47
48    fn release_decision(&mut self) {
49        if let Some(to_release) = self.to_release.take() {
50            for item in to_release {
51                self.output.send(item).unwrap();
52            }
53        } else {
54            panic!("No decision to release");
55        }
56    }
57}
