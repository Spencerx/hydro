27#[sealed::sealed]
28/// A trait implemented by closures that can instantiate a compiled simulation.
29///
30/// This is needed to ensure [`RefUnwindSafe`] so instances can be created during fuzzing.
31pub trait Instantiator<'a>: RefUnwindSafe + Fn() -> CompiledSimInstance<'a> {}
32#[sealed::sealed]
33impl<'a, T: RefUnwindSafe + Fn() -> CompiledSimInstance<'a>> Instantiator<'a> for T {}
34
35type SimLoaded<'a> = libloading::Symbol<
36    'a,
37    unsafe extern "Rust" fn(
38        HashMap<usize, UnboundedSender<Bytes>>,
39        HashMap<usize, UnboundedReceiverStream<Bytes>>,
40    ) -> (
41        Dfir<'static>,
42        Vec<(&'static str, Dfir<'static>)>,
43        HashMap<&'static str, Vec<Box<dyn SimHook>>>,
44    ),
45>;
46
47impl CompiledSim {
48    /// Executes the given closure with a single instance of the compiled simulation.
49    pub fn with_instance<T>(&self, thunk: impl FnOnce(CompiledSimInstance) -> T) -> T {
50        self.with_instantiator(|instantiator| thunk(instantiator()))
51    }
52
53    /// Executes the given closure with an [`Instantiator`], which can be called to create
54    /// independent instances of the simulation. This is useful for fuzzing, where we need to
55    /// re-execute the simulation several times with different decisions.
56    pub fn with_instantiator<T>(&self, thunk: impl FnOnce(&dyn Instantiator) -> T) -> T {
57        let func: SimLoaded = unsafe { self.lib.get(b"__hydro_runtime").unwrap() };
58        thunk(
59            &(|| CompiledSimInstance {
60                func: func.clone(),
61                remaining_ports: self.external_ports.iter().cloned().collect(),
62                input_ports: HashMap::new(),
63                output_ports: HashMap::new(),
64            }),
65        )
66    }
67
68    /// Uses a fuzzing strategy to explore possible executions of the simulation. The provided
69    /// closure will be repeatedly executed with instances of the Hydro program where the
70    /// batching boundaries, order of messages, and retries are varied.
71    ///
72    /// During development, you should run the test that invokes this function with the `cargo sim`
73    /// command, which will use `libfuzzer` to intelligently explore the execution space. If a
74    /// failure is found, a minimized test case will be produced in a `sim-failures` directory.
75    /// When running the test with `cargo test` (such as in CI), if a reproducer is found it will
76    /// be executed, and if no reproducer is found a small number of random executions will be
77    /// performed.
78    pub fn fuzz<'a>(&'a self, thunk: impl AsyncFn(CompiledSimInstance) + RefUnwindSafe) {
79        let caller_fn = crate::compile::ir::backtrace::Backtrace::get_backtrace(0)
80            .elements()
81            .into_iter()
82            .find(|e| {
83                !e.fn_name.starts_with("hydro_lang::sim::compiled")
84                    && !e.fn_name.starts_with("hydro_lang::sim::flow")
85                    && !e.fn_name.starts_with("fuzz<")
86            })
87            .unwrap();
88
89        let caller_path = Path::new(&caller_fn.filename.unwrap()).to_path_buf();
90        let repro_folder = caller_path.parent().unwrap().join("sim-failures");
91
92        let caller_fuzz_repro_path = repro_folder
93            .join(caller_fn.fn_name.replace("::", "__"))
94            .with_extension("bin");
95
96        if std::env::var("BOLERO_FUZZER").is_ok() {
97            let corpus_dir = std::env::current_dir().unwrap().join(".fuzz-corpus");
98            std::fs::create_dir_all(&corpus_dir).unwrap();
99            let libfuzzer_args = format!(
100                "{} {} -artifact_prefix={}/ -handle_abrt=0",
101                corpus_dir.to_str().unwrap(),
102                corpus_dir.to_str().unwrap(),
103                corpus_dir.to_str().unwrap(),
104            );
105
106            std::fs::create_dir_all(&repro_folder).unwrap();
107
108            unsafe {
109                std::env::set_var(
110                    "BOLERO_FAILURE_OUTPUT",
111                    caller_fuzz_repro_path.to_str().unwrap(),
112                );
113
114                std::env::set_var("BOLERO_LIBFUZZER_ARGS", libfuzzer_args);
115            }
116
117            self.with_instantiator(|instantiator| {
118                bolero::test(bolero::TargetLocation {
119                    package_name: "",
120                    manifest_dir: "",
121                    module_path: "",
122                    file: "",
123                    line: 0,
124                    item_path: "<unknown>::__bolero_item_path__",
125                    test_name: None,
126                })
127                .run(move || {
128                    let instance = instantiator();
129                    tokio::runtime::Builder::new_current_thread()
130                        .build()
131                        .unwrap()
132                        .block_on(async {
133                            let local_set = tokio::task::LocalSet::new();
134                            local_set.run_until(thunk(instance)).await
135                        })
136                })
137            });
138        } else if let Ok(existing_bytes) = std::fs::read(&caller_fuzz_repro_path) {
139            self.with_instance(|instance| {
140                bolero::bolero_engine::any::scope::with(
141                    Box::new(bolero::bolero_engine::driver::object::Object(
142                        bolero::bolero_engine::driver::bytes::Driver::new(
143                            existing_bytes,
144                            &Default::default(),
145                        ),
146                    )),
147                    || {
148                        tokio::runtime::Builder::new_current_thread()
149                            .build()
150                            .unwrap()
151                            .block_on(async {
152                                let local_set = tokio::task::LocalSet::new();
153                                local_set.run_until(thunk(instance)).await
154                            })
155                    },
156                )
157            });
158        } else {
159            eprintln!(
160                "Running a fuzz test without `cargo sim` and no reproducer found at {}, defaulting to 8192 iterations with random inputs.",
161                caller_fuzz_repro_path.display()
162            );
163            self.with_instantiator(|instantiator| {
164                bolero::test(bolero::TargetLocation {
165                    package_name: "",
166                    manifest_dir: "",
167                    module_path: "",
168                    file: ".",
169                    line: 0,
170                    item_path: "<unknown>::__bolero_item_path__",
171                    test_name: None,
172                })
173                .with_iterations(8192)
174                .run(move || {
175                    let instance = instantiator();
176                    tokio::runtime::Builder::new_current_thread()
177                        .build()
178                        .unwrap()
179                        .block_on(async {
180                            let local_set = tokio::task::LocalSet::new();
181                            local_set.run_until(thunk(instance)).await
182                        })
183                })
184            });
185        }
186    }
187
188    /// Exhaustively searches all possible executions of the simulation. The provided
189    /// closure will be repeatedly executed with instances of the Hydro program where the
190    /// batching boundaries, order of messages, and retries are varied.
191    ///
192    /// Exhaustive searching is feasible when the inputs to the Hydro program are finite and there
193    /// are no dataflow loops that generate infinite messages. Exhaustive searching provides a
194    /// stronger guarantee of correctness than fuzzing, but may take a long time to complete.
195    /// Because no fuzzer is involved, you can run exhaustive tests with `cargo test`.
196    pub fn exhaustive<'a>(&'a self, thunk: impl AsyncFn(CompiledSimInstance) + RefUnwindSafe) {
197        if std::env::var("BOLERO_FUZZER").is_ok() {
198            eprintln!(
199                "Cannot run exhaustive tests with a fuzzer. Please use `cargo test` instead of `cargo sim`."
200            );
201            std::process::abort();
202        }
203
204        self.with_instantiator(|instantiator| {
205            bolero::test(bolero::TargetLocation {
206                package_name: "",
207                manifest_dir: "",
208                module_path: "",
209                file: "",
210                line: 0,
211                item_path: "<unknown>::__bolero_item_path__",
212                test_name: None,
213            })
214            .exhaustive()
215            .run(move || {
216                let instance = instantiator();
217                tokio::runtime::Builder::new_current_thread()
218                    .build()
219                    .unwrap()
220                    .block_on(async {
221                        let local_set = tokio::task::LocalSet::new();
222                        local_set.run_until(thunk(instance)).await;
223                    })
224            })
225        });
226    }
227}

229/// A single instance of a compiled Hydro simulation, which provides methods to interactively
230/// execute the simulation, feed inputs, and receive outputs.
231pub struct CompiledSimInstance<'a> {
232    func: SimLoaded<'a>,
233    remaining_ports: HashSet<usize>,
234    output_ports: HashMap<usize, UnboundedSender<Bytes>>,
235    input_ports: HashMap<usize, UnboundedReceiverStream<Bytes>>,
236}
237
238impl<'a> CompiledSimInstance<'a> {
239    /// Like the corresponding method on [`crate::compile::deploy::DeployResult`], connects to the
240    /// given input port, and returns a closure that can be used to send messages to it.
241    pub fn connect_sink_bincode<T: 'static + Send + Serialize + DeserializeOwned>(
242        &mut self,
243        port: &ExternalBincodeSink<T>,
244    ) -> impl Fn(T) -> Result<(), tokio::sync::mpsc::error::SendError<Bytes>> + 'a {
245        assert!(self.remaining_ports.remove(&port.port_id));
246        let (sink, source) = dfir_rs::util::unbounded_channel::<Bytes>();
247        self.input_ports.insert(port.port_id, source);
248        move |t| sink.send(bincode::serialize(&t).unwrap().into())
249    }
250
251    /// Like the corresponding method on [`crate::compile::deploy::DeployResult`], connects to the
252    /// given output port, and returns a stream that can be used to receive messages from it.
253    pub fn connect_source_bincode<T: 'static + Send + Serialize + DeserializeOwned>(
254        &mut self,
255        port: &ExternalBincodeStream<T>,
256    ) -> impl Stream<Item = T> + 'a {
257        assert!(self.remaining_ports.remove(&port.port_id));
258        let (sink, source) = dfir_rs::util::unbounded_channel::<Bytes>();
259        self.output_ports.insert(port.port_id, sink);
260        source.map(|b| bincode::deserialize(&b).unwrap())
261    }
262
263    /// Launches the simulation, which will asynchronously simulate the Hydro program. This should
264    /// be invoked after connecting all inputs and outputs, but before receiving any messages.
265    pub fn launch(self) {
266        if !self.remaining_ports.is_empty() {
267            panic!(
268                "Cannot launch DFIR because some of the inputs / outputs have not been connected."
269            )
270        }
271
272        let (async_dfir, ticks, hooks) =
273            unsafe { (self.func)(self.output_ports, self.input_ports) };
274        let mut launched = LaunchedSim {
275            async_dfir,
276            possibly_ready_ticks: vec![],
277            not_ready_ticks: ticks.into_iter().collect(),
278            hooks,
279        };
280
281        tokio::task::spawn_local(async move {
282            launched.scheduler().await;
283        });
284    }
285}

287/// A running simulation, which manages the async DFIR and tick DFIRs, and makes decisions
288/// about scheduling ticks and choices for non-deterministic operators like batch.
289struct LaunchedSim {
290    async_dfir: Dfir<'static>,
291    possibly_ready_ticks: Vec<(&'static str, Dfir<'static>)>,
292    not_ready_ticks: Vec<(&'static str, Dfir<'static>)>,
293    hooks: HashMap<&'static str, Vec<Box<dyn SimHook>>>,
294}
295
296impl LaunchedSim {
297    async fn scheduler(&mut self) {
298        loop {
299            tokio::task::yield_now().await;
300            if self.async_dfir.run_available().await {
301                self.possibly_ready_ticks.append(&mut self.not_ready_ticks);
302                continue;
303            } else {
304                use bolero::generator::*;
305
306                let (ready, mut not_ready): (Vec<_>, Vec<_>) =
307                    self.possibly_ready_ticks.drain(..).partition(|(name, _)| {
308                        self.hooks.get(name).unwrap().iter().any(|hook| {
309                            hook.current_decision().unwrap_or(false)
310                                || hook.can_make_nontrivial_decision()
311                        })
312                    });
313
314                self.possibly_ready_ticks = ready;
315                self.not_ready_ticks.append(&mut not_ready);
316
317                if self.possibly_ready_ticks.is_empty() {
318                    break;
319                } else {
320                    let next_tick = (0..self.possibly_ready_ticks.len()).any();
321                    let mut removed: (&'static str, Dfir<'static>) =
322                        self.possibly_ready_ticks.remove(next_tick);
323
324                    let hooks = self.hooks.get_mut(removed.0).unwrap();
325                    let mut remaining_decision_count = hooks.len();
326                    let mut made_nontrivial_decision = false;
327
328                    // first, scan manual decisions
329                    hooks.iter_mut().for_each(|hook| {
330                        if let Some(is_nontrivial) = hook.current_decision() {
331                            made_nontrivial_decision |= is_nontrivial;
332                            remaining_decision_count -= 1;
333                        }
334                    });
335
336                    bolero_generator::any::scope::borrow_with(|driver| {
337                        hooks.iter_mut().for_each(|hook| {
338                            if hook.current_decision().is_none() {
339                                made_nontrivial_decision |= hook.autonomous_decision(
340                                    driver,
341                                    !made_nontrivial_decision && remaining_decision_count == 1,
342                                );
343                                remaining_decision_count -= 1;
344                            }
345
346                            hook.release_decision();
347                        });
348                    });
349
350                    assert!(removed.1.run_tick().await);
351                    self.possibly_ready_ticks.push(removed);
352                }
353            }
354        }
355    }
